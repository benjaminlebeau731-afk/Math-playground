import React, { useState, useEffect, useRef } from 'react';
import { 
  Search, 
  Mic, 
  Flame, 
  History, 
  Play, 
  Pause, 
  X, 
  Menu, 
  Activity,
  Sparkles,
  CalendarDays,
  Youtube,
  Settings,
  Moon,
  Sun,
  Trash2,
  RefreshCw
} from 'lucide-react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInAnonymously, onAuthStateChanged } from 'firebase/auth';
import { 
  getFirestore, 
  collection, 
  addDoc, 
  onSnapshot, 
  query, 
  serverTimestamp,
  deleteDoc,
  doc,
  getDocs,
  writeBatch
} from 'firebase/firestore';

// --- Firebase Configuration ---
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'fire-ai-v1';

const NOTES = ["C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "A#", "B"];

export default function App() {
  const [user, setUser] = useState(null);
  const [activeTab, setActiveTab] = useState('newreleases'); // Set default to releases to check fix
  const [searchQuery, setSearchQuery] = useState('');
  const [isListening, setIsListening] = useState(false);
  const [isSearching, setIsSearching] = useState(false);
  const [searchResults, setSearchResults] = useState([]);
  const [newReleases, setNewReleases] = useState([]);
  const [isLoadingReleases, setIsLoadingReleases] = useState(false);
  const [forYouSongs, setForYouSongs] = useState([]);
  const [history, setHistory] = useState([]);
  const [isSidebarOpen, setSidebarOpen] = useState(false);
  const [darkMode, setDarkMode] = useState(false);
  
  const [detectedPitch, setDetectedPitch] = useState(null);
  const [detectedNote, setDetectedNote] = useState("");
  const melodyBuffer = useRef([]); 
  const audioContextRef = useRef(null);
  const analyserRef = useRef(null);
  const animationFrameRef = useRef(null);

  const [currentTrack, setCurrentTrack] = useState(null);
  const [isPlaying, setIsPlaying] = useState(false);
  const [progress, setProgress] = useState(0);
  const audioRef = useRef(null);

  useEffect(() => {
    const initAuth = async () => {
      try {
        await signInAnonymously(auth);
      } catch (err) {
        console.error("Auth failed:", err);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, (u) => setUser(u));
    return () => unsubscribe();
  }, []);

  const fetchNewReleases = async () => {
    setIsLoadingReleases(true);
    try {
      // Added a timestamp to the URL to bypass browser caching and get fresh data
      const cacheBuster = new Date().getTime();
      const response = await fetch(`https://itunes.apple.com/us/rss/topsongs/limit=24/json?cb=${cacheBuster}`);
      const data = await response.json();
      const formatted = data.feed.entry.map(entry => ({
        id: entry.id.attributes['im:id'],
        title: entry['im:name'].label,
        artist: entry['im:artist'].label,
        artwork: entry['im:image'][2].label,
        previewUrl: entry.link[1]?.attributes.href,
        duration: "3:30" 
      }));
      setNewReleases(formatted);
    } catch (err) {
      console.error("Failed to fetch new releases", err);
    } finally {
      setIsLoadingReleases(false);
    }
  };

  useEffect(() => {
    fetchNewReleases();
  }, []);

  useEffect(() => {
    const generateRecommendations = async () => {
      if (history.length > 0) {
        const artists = history.map(h => h.artist);
        const mostFrequentArtist = artists.sort((a,b) =>
          artists.filter(v => v === a).length - artists.filter(v => v === b).length
        ).pop();

        try {
          const response = await fetch(`https://itunes.apple.com/search?term=${encodeURIComponent(mostFrequentArtist)}&entity=song&limit=8`);
          const data = await response.json();
          const formatted = data.results.map(item => ({
            id: item.trackId.toString(),
            title: item.trackName,
            artist: item.artistName,
            artwork: item.artworkUrl100.replace('100x100', '400x400'),
            previewUrl: item.previewUrl
          }));
          setForYouSongs(formatted);
        } catch (err) {
          console.error("Recommendation error:", err);
        }
      } else {
        setForYouSongs(newReleases.slice(0, 8));
      }
    };
    generateRecommendations();
  }, [history, newReleases]);

  useEffect(() => {
    if (!user) return;
    const q = collection(db, 'artifacts', appId, 'users', user.uid, 'history');
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const docs = snapshot.docs.map(doc => ({ docId: doc.id, ...doc.data() }));
      const uniqueHistory = [];
      const seenIds = new Set();
      docs.sort((a, b) => (b.timestamp?.seconds || 0) - (a.timestamp?.seconds || 0)).forEach(item => {
        if (!seenIds.has(item.id)) {
          seenIds.add(item.id);
          uniqueHistory.push(item);
        }
      });
      setHistory(uniqueHistory);
    }, (err) => console.error("Firestore error:", err));
    return () => unsubscribe();
  }, [user]);

  useEffect(() => {
    if (audioRef.current) {
      if (isPlaying) {
        audioRef.current.play().catch(e => console.log("Audio play blocked", e));
      } else {
        audioRef.current.pause();
      }
    }
  }, [isPlaying, currentTrack]);

  const addToHistory = async (song) => {
    if (!user) return;
    try {
      await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'history'), {
        ...song,
        timestamp: serverTimestamp(),
        type: 'play'
      });
    } catch (err) {
      console.error("Failed to add to history:", err);
    }
  };

  const deleteFromHistory = async (docId) => {
    if (!user || !docId) return;
    try {
      await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'history', docId));
    } catch (err) {
      console.error("Failed to delete item:", err);
    }
  };

  const clearAllHistory = async () => {
    if (!user) return;
    try {
      const q = collection(db, 'artifacts', appId, 'users', user.uid, 'history');
      const snapshot = await getDocs(q);
      const batch = writeBatch(db);
      snapshot.docs.forEach((doc) => {
        batch.delete(doc.ref);
      });
      await batch.commit();
    } catch (err) {
      console.error("Failed to clear history:", err);
    }
  };

  const handleTrackSelect = (song) => {
    if (currentTrack?.id === song.id) {
      setIsPlaying(!isPlaying);
    } else {
      setCurrentTrack(song);
      setIsPlaying(true);
      setProgress(0);
      addToHistory(song);
    }
  };

  const getYouTubeLink = (song) => {
    return `https://www.youtube.com/results?search_query=${encodeURIComponent(song.artist + ' ' + song.title)}`;
  };

  const getNoteFromFrequency = (frequency) => {
    const noteNum = 12 * (Math.log(frequency / 440) / Math.log(2));
    const roundedNote = Math.round(noteNum) + 69;
    const octave = Math.floor(roundedNote / 12) - 1;
    const noteName = NOTES[roundedNote % 12];
    return `${noteName}${octave}`;
  };

  const autoCorrelate = (buffer, sampleRate) => {
    let SIZE = buffer.length;
    let rms = 0;
    for (let i = 0; i < SIZE; i++) {
      let val = buffer[i];
      rms += val * val;
    }
    rms = Math.sqrt(rms / SIZE);
    if (rms < 0.01) return -1;
    let r1 = 0, r2 = SIZE - 1, thres = 0.2;
    for (let i = 0; i < SIZE / 2; i++) { if (Math.abs(buffer[i]) < thres) { r1 = i; break; } }
    for (let i = 1; i < SIZE / 2; i++) { if (Math.abs(buffer[SIZE - i]) < thres) { r2 = SIZE - i; break; } }
    const trimmedBuffer = buffer.slice(r1, r2);
    const trimmedSize = trimmedBuffer.length;
    let c = new Array(trimmedSize).fill(0);
    for (let i = 0; i < trimmedSize; i++) {
      for (let j = 0; j < trimmedSize - i; j++) {
        c[i] = c[i] + trimmedBuffer[j] * trimmedBuffer[j + i];
      }
    }
    let d = 0;
    while (c[d] > c[d + 1]) d++;
    let maxval = -1, maxpos = -1;
    for (let i = d; i < trimmedSize; i++) {
      if (c[i] > maxval) { maxval = c[i]; maxpos = i; }
    }
    return sampleRate / maxpos;
  };

  const updatePitch = () => {
    if (!analyserRef.current || !isListening) return;
    const buffer = new Float32Array(2048);
    analyserRef.current.getFloatTimeDomainData(buffer);
    const ac = autoCorrelate(buffer, audioContextRef.current.sampleRate);
    if (ac !== -1 && ac > 50 && ac < 1200) { 
      setDetectedPitch(Math.round(ac));
      const note = getNoteFromFrequency(ac);
      setDetectedNote(note);
      melodyBuffer.current.push(note);
    }
    animationFrameRef.current = requestAnimationFrame(updatePitch);
  };

  const startListening = async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
      audioContextRef.current = new (window.AudioContext || window.webkitAudioContext)();
      
      if (audioContextRef.current.state === 'suspended') {
        await audioContextRef.current.resume();
      }

      analyserRef.current = audioContextRef.current.createAnalyser();
      analyserRef.current.fftSize = 2048;
      const source = audioContextRef.current.createMediaStreamSource(stream);
      source.connect(analyserRef.current);
      melodyBuffer.current = [];
      setIsListening(true);
      updatePitch();
      
      setTimeout(() => stopListening(true), 6000);
    } catch (err) {
      console.error("Microphone access denied or error:", err);
    }
  };

  const stopListening = (shouldSearch = false) => {
    setIsListening(false);
    if (animationFrameRef.current) cancelAnimationFrame(animationFrameRef.current);
    if (audioContextRef.current) audioContextRef.current.close();
    
    if (shouldSearch && melodyBuffer.current.length > 10) {
      const pureNotes = melodyBuffer.current.map(n => n.replace(/\d+/g, ''));
      const pureMap = {};
      pureNotes.forEach(n => pureMap[n] = (pureMap[n] || 0) + 1);

      const topPureNotes = Object.entries(pureMap)
        .sort((a, b) => b[1] - a[1])
        .slice(0, 3)
        .map(n => n[0])
        .join(" ");
      
      performGlobalSearch(`${topPureNotes} melody hits`);
    }
  };

  const performGlobalSearch = async (term) => {
    if (!term.trim()) return;
    setIsSearching(true);
    try {
      const response = await fetch(`https://itunes.apple.com/search?term=${encodeURIComponent(term)}&entity=song&limit=20`);
      const data = await response.json();
      const formatted = data.results.map(item => ({
        id: item.trackId.toString(),
        title: item.trackName,
        artist: item.artistName,
        album: item.collectionName,
        artwork: item.artworkUrl100.replace('100x100', '400x400'),
        previewUrl: item.previewUrl
      }));
      setSearchResults(formatted);
    } catch (err) {
      console.error(err);
    } finally {
      setIsSearching(false);
      setActiveTab('search');
    }
  };

  const navigate = (tab) => {
    setActiveTab(tab);
    setSidebarOpen(false);
  };

  return (
    <div className={`flex h-screen font-sans overflow-hidden transition-colors duration-300 ${darkMode ? 'bg-slate-950 text-slate-100' : 'bg-slate-50 text-slate-900'}`}>
      <style>{`
        body { font-family: 'Inter', system-ui, -apple-system, sans-serif; }
      `}</style>

      {currentTrack?.previewUrl && (
        <audio 
          ref={audioRef} 
          src={currentTrack.previewUrl} 
          onTimeUpdate={() => setProgress((audioRef.current.currentTime / audioRef.current.duration) * 100)}
          onEnded={() => setIsPlaying(false)}
        />
      )}

      {/* Mobile Sidebar Overlay */}
      {isSidebarOpen && (
        <div 
          className="fixed inset-0 bg-black/40 backdrop-blur-sm z-40 md:hidden transition-opacity animate-in fade-in" 
          onClick={() => setSidebarOpen(false)}
        />
      )}

      {/* Sidebar */}
      <aside className={`fixed inset-y-0 left-0 z-50 w-64 border-r transform transition-transform duration-300 ease-in-out md:relative md:translate-x-0 ${darkMode ? 'bg-slate-900 border-slate-800' : 'bg-white border-slate-200'} ${isSidebarOpen ? 'translate-x-0 shadow-2xl' : '-translate-x-full'}`}>
        <div className="p-6 h-full flex flex-col">
          <div className="flex items-center justify-between mb-8">
            <div className="flex items-center space-x-2.5">
              <div className="bg-gradient-to-br from-orange-400 to-red-600 p-2 rounded-xl text-white shadow-sm">
                 <Flame size={20} fill="currentColor" />
              </div>
              <h1 className="text-xl font-bold tracking-tight">Fire AI</h1>
            </div>
            <button onClick={() => setSidebarOpen(false)} className="md:hidden p-2 text-slate-400 hover:text-slate-600">
              <X size={20} />
            </button>
          </div>
          <nav className="space-y-1.5 flex-1">
            <button onClick={() => { navigate('search'); setSearchQuery(''); setSearchResults([]); }} className={`flex items-center space-x-3 w-full p-2.5 rounded-lg transition-all ${activeTab === 'search' ? 'bg-orange-500 text-white shadow-sm' : `${darkMode ? 'text-slate-400 hover:bg-slate-800' : 'text-slate-600 hover:bg-slate-100'}`}`}>
              <Search size={18} /> <span className="text-sm font-medium">Search</span>
            </button>
            <button onClick={() => navigate('newreleases')} className={`flex items-center space-x-3 w-full p-2.5 rounded-lg transition-all ${activeTab === 'newreleases' ? 'bg-orange-500 text-white shadow-sm' : `${darkMode ? 'text-slate-400 hover:bg-slate-800' : 'text-slate-600 hover:bg-slate-100'}`}`}>
              <CalendarDays size={18} /> <span className="text-sm font-medium">Hot Releases</span>
            </button>
            <button onClick={() => navigate('foryou')} className={`flex items-center space-x-3 w-full p-2.5 rounded-lg transition-all ${activeTab === 'foryou' ? 'bg-orange-500 text-white shadow-sm' : `${darkMode ? 'text-slate-400 hover:bg-slate-800' : 'text-slate-600 hover:bg-slate-100'}`}`}>
              <Sparkles size={18} /> <span className="text-sm font-medium">Ignited For You</span>
            </button>
            <button onClick={() => navigate('listen')} className={`flex items-center space-x-3 w-full p-2.5 rounded-lg transition-all ${activeTab === 'listen' ? 'bg-orange-500 text-white shadow-sm' : `${darkMode ? 'text-slate-400 hover:bg-slate-800' : 'text-slate-600 hover:bg-slate-100'}`}`}>
              <Mic size={18} /> <span className="text-sm font-medium">Melody Ignite</span>
            </button>
            <button onClick={() => navigate('history')} className={`flex items-center space-x-3 w-full p-2.5 rounded-lg transition-all ${activeTab === 'history' ? 'bg-orange-500 text-white shadow-sm' : `${darkMode ? 'text-slate-400 hover:bg-slate-800' : 'text-slate-600 hover:bg-slate-100'}`}`}>
              <History size={18} /> <span className="text-sm font-medium">My Burn Log</span>
            </button>
          </nav>

          <button onClick={() => navigate('settings')} className={`flex items-center space-x-3 w-full p-2.5 rounded-lg transition-all mb-4 ${activeTab === 'settings' ? 'bg-orange-500 text-white' : `${darkMode ? 'text-slate-400 hover:bg-slate-800' : 'text-slate-600 hover:bg-slate-100'}`}`}>
            <Settings size={18} /> <span className="text-sm font-medium">Settings</span>
          </button>
        </div>
      </aside>

      <main className="flex-1 flex flex-col min-w-0 overflow-hidden relative">
        <header className={`h-16 flex items-center px-4 md:px-6 border-b sticky top-0 z-30 transition-colors duration-300 ${darkMode ? 'bg-slate-950/80 border-slate-800' : 'bg-white/80 border-slate-200'} backdrop-blur-md`}>
          <button 
            onClick={() => setSidebarOpen(true)} 
            className={`p-2 mr-2 rounded-lg md:hidden transition-colors ${darkMode ? 'text-slate-300 hover:bg-slate-800' : 'text-slate-600 hover:bg-slate-100'}`}
          >
            <Menu size={22} />
          </button>
          
          <div className="flex-1 max-w-xl">
            <div className="relative">
              <Search className="absolute left-3 top-1/2 -translate-y-1/2 text-slate-400" size={16} />
              <input
                type="text"
                placeholder="Find tracks..."
                className={`w-full pl-9 pr-10 py-1.5 md:py-2 rounded-xl text-sm outline-none focus:ring-2 focus:ring-orange-500/20 transition-all ${darkMode ? 'bg-slate-900 border-slate-800 text-white' : 'bg-slate-100 border-transparent text-slate-900'}`}
                value={searchQuery}
                onChange={(e) => setSearchQuery(e.target.value)}
                onKeyDown={(e) => {
                  if (e.key === 'Enter') {
                    setActiveTab('search');
                    performGlobalSearch(searchQuery);
                  }
                }}
              />
            </div>
          </div>
          <div className="ml-auto flex items-center space-x-1">
             <button 
               onClick={() => setDarkMode(!darkMode)}
               className={`p-2 rounded-full transition-colors ${darkMode ? 'hover:bg-slate-800 text-yellow-400' : 'hover:bg-slate-100 text-slate-500'}`}
             >
               {darkMode ? <Sun size={20} /> : <Moon size={20} />}
             </button>
          </div>
        </header>

        <div className="flex-1 overflow-y-auto p-4 md:p-6 pb-32">
          {activeTab === 'search' && (
            <div className="max-w-5xl mx-auto">
              {isSearching ? (
                 <div className="h-64 flex flex-col items-center justify-center space-y-4">
                    <div className="w-12 h-12 border-4 border-orange-500/20 border-t-orange-500 rounded-full animate-spin"></div>
                    <p className="text-sm font-medium text-slate-400">Igniting search results...</p>
                 </div>
              ) : searchResults.length > 0 ? (
                <div>
                  <h2 className="text-xl font-bold mb-4 tracking-tight">Search Results</h2>
                  <div className="space-y-1.5">
                    {searchResults.map(song => (
                      <div key={song.id} className={`group flex items-center p-2.5 rounded-xl border transition-all cursor-pointer ${darkMode ? 'bg-slate-900 border-slate-800 hover:border-orange-500/50' : 'bg-white border-slate-200 hover:border-orange-500/30'} ${currentTrack?.id === song.id ? 'ring-1 ring-orange-500 border-orange-500' : ''}`}>
                        <div onClick={() => handleTrackSelect(song)} className="flex-1 flex items-center min-w-0">
                          <div className="w-11 h-11 rounded-lg overflow-hidden mr-4 flex-shrink-0 bg-slate-800">
                            <img src={song.artwork} className="w-full h-full object-cover" alt="" />
                          </div>
                          <div className="flex-1 min-w-0">
                            <p className="font-semibold text-sm truncate">{song.title}</p>
                            <p className={`text-xs truncate ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>{song.artist}</p>
                          </div>
                        </div>
                        <div className="flex items-center space-x-2 px-2">
                          <a href={getYouTubeLink(song)} target="_blank" rel="noopener noreferrer" className="p-2 text-slate-400 hover:text-red-600 transition-colors" title="Listen on YouTube" onClick={(e) => e.stopPropagation()}>
                            <Youtube size={18} />
                          </a>
                        </div>
                      </div>
                    ))}
                  </div>
                </div>
              ) : (
                <div className="h-full flex flex-col items-center justify-center text-center py-24 opacity-30">
                   <Flame size={48} className="mb-4" />
                   <p className="text-base font-medium">Search for Fire tracks</p>
                </div>
              )}
            </div>
          )}

          {activeTab === 'newreleases' && (
            <div className="max-w-5xl mx-auto">
              <div className="flex items-center justify-between mb-2">
                <div className="flex items-center space-x-2.5">
                  <Flame className="text-orange-500" size={22} fill="currentColor" />
                  <h2 className="text-xl font-bold tracking-tight">Hot Releases</h2>
                </div>
                <button 
                  onClick={fetchNewReleases}
                  disabled={isLoadingReleases}
                  className={`p-2 rounded-lg transition-colors ${darkMode ? 'hover:bg-slate-800 text-slate-400' : 'hover:bg-slate-100 text-slate-500'} ${isLoadingReleases ? 'animate-spin opacity-50' : ''}`}
                >
                  <RefreshCw size={18} />
                </button>
              </div>
              <p className={`mb-8 text-sm ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>The latest tracks catching fire right now.</p>
              
              {isLoadingReleases ? (
                 <div className="h-64 flex flex-col items-center justify-center space-y-4">
                    <div className="w-10 h-10 border-4 border-orange-500/20 border-t-orange-500 rounded-full animate-spin"></div>
                    <p className="text-xs font-medium text-slate-400">Refreshing list...</p>
                 </div>
              ) : (
                <div className="grid grid-cols-2 sm:grid-cols-3 lg:grid-cols-4 xl:grid-cols-6 gap-3 md:gap-5 animate-in fade-in duration-500">
                  {newReleases.map(song => (
                    <div key={song.id} className="group flex flex-col">
                      <div onClick={() => handleTrackSelect(song)} className="relative aspect-square rounded-2xl overflow-hidden mb-2.5 shadow-sm group-hover:shadow-md transition-all duration-200 cursor-pointer">
                        <img src={song.artwork} className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-300" alt="" />
                        <div className="absolute inset-0 bg-black/40 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center">
                          <div className="bg-white p-3 rounded-full text-orange-600 scale-90 group-hover:scale-100 transition-transform">
                            {currentTrack?.id === song.id && isPlaying ? <Pause size={20} fill="currentColor" /> : <Play size={20} fill="currentColor" className="ml-0.5" />}
                          </div>
                        </div>
                      </div>
                      <div className="flex justify-between items-start">
                        <div className="min-w-0 pr-2">
                          <p className="font-semibold text-[11px] md:text-xs truncate group-hover:text-orange-500 transition-colors">{song.title}</p>
                          <p className={`text-[10px] truncate ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>{song.artist}</p>
                        </div>
                      </div>
                    </div>
                  ))}
                </div>
              )}
            </div>
          )}

          {activeTab === 'foryou' && (
            <div className="max-w-5xl mx-auto">
              <div className="flex items-center space-x-2.5 mb-2">
                <Sparkles className="text-orange-500" size={22} fill="currentColor" />
                <h2 className="text-xl font-bold tracking-tight">Ignited For You</h2>
              </div>
              <p className={`mb-8 text-sm ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>Hand-picked fire tunes based on your burn log.</p>
              
              <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                {forYouSongs.map((song, idx) => (
                  <div key={song.id} className={`flex items-center p-3 rounded-2xl transition-all border group ${currentTrack?.id === song.id ? 'bg-orange-500 text-white border-orange-500 shadow-md' : `${darkMode ? 'bg-slate-900 border-slate-800' : 'bg-white border-slate-100'}`}`}>
                    <div onClick={() => handleTrackSelect(song)} className="flex items-center flex-1 cursor-pointer min-w-0">
                      <div className="text-lg font-bold opacity-20 mr-4 w-5">{idx + 1}</div>
                      <img src={song.artwork} className="w-14 h-14 rounded-xl shadow-sm mr-4 object-cover" alt="" />
                      <div className="flex-1 min-w-0">
                        <p className="font-semibold text-sm truncate">{song.title}</p>
                        <p className={`text-xs truncate ${currentTrack?.id === song.id ? 'text-orange-100' : 'text-slate-400'}`}>{song.artist}</p>
                      </div>
                    </div>
                    <div className="flex items-center space-x-3 ml-4">
                      <a href={getYouTubeLink(song)} target="_blank" rel="noopener noreferrer" className={`${currentTrack?.id === song.id ? 'text-white' : 'text-slate-400 hover:text-red-500'}`}>
                        <Youtube size={18} />
                      </a>
                      <div onClick={() => handleTrackSelect(song)} className={`p-2 rounded-full cursor-pointer ${currentTrack?.id === song.id ? 'bg-white/20 text-white' : 'bg-orange-50 text-orange-600'}`}>
                        {currentTrack?.id === song.id && isPlaying ? <Pause size={16} fill="currentColor" /> : <Play size={16} fill="currentColor" className="ml-0.5" />}
                      </div>
                    </div>
                  </div>
                ))}
              </div>
            </div>
          )}

          {activeTab === 'history' && (
            <div className="max-w-3xl mx-auto">
              <div className="flex items-center justify-between mb-6">
                <h2 className="text-xl font-bold tracking-tight">My Burn Log</h2>
                {history.length > 0 && (
                  <button 
                    onClick={clearAllHistory}
                    className="flex items-center space-x-2 text-[11px] font-bold text-red-500 hover:text-red-600 transition-colors uppercase tracking-widest"
                  >
                    <Trash2 size={12} />
                    <span>Clear Log</span>
                  </button>
                )}
              </div>
              {history.length === 0 ? (
                <div className="py-24 text-center text-slate-400">
                  <Flame size={40} className="mx-auto mb-4 opacity-10" />
                  <p className="text-sm">Your log is empty. Go find some fire.</p>
                </div>
              ) : (
                <div className="grid gap-2">
                  {history.map(item => (
                    <div key={item.docId} className={`flex items-center p-3 rounded-xl border transition-all cursor-pointer group ${darkMode ? 'bg-slate-900 border-slate-800 hover:border-orange-500/50' : 'bg-white border-slate-100 hover:border-orange-500/20'}`}>
                      <div onClick={() => handleTrackSelect(item)} className="flex-1 flex items-center min-w-0">
                        <div className="w-10 h-10 rounded-lg overflow-hidden mr-4 bg-slate-800">
                          <img src={item.artwork} className="w-full h-full object-cover" alt="" />
                        </div>
                        <div className="flex-1 min-w-0">
                          <p className="font-semibold text-sm truncate">{item.title}</p>
                          <p className={`text-xs truncate ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>{item.artist}</p>
                        </div>
                      </div>
                      <div className="flex items-center space-x-2">
                        <a href={getYouTubeLink(item)} target="_blank" rel="noopener noreferrer" className="p-2 text-slate-400 hover:text-red-600">
                          <Youtube size={18} />
                        </a>
                        <button 
                          onClick={(e) => { e.stopPropagation(); deleteFromHistory(item.docId); }}
                          className="p-2 text-slate-400 hover:text-red-500 opacity-0 group-hover:opacity-100 transition-all"
                        >
                          <X size={16} />
                        </button>
                      </div>
                    </div>
                  ))}
                </div>
              )}
            </div>
          )}

          {activeTab === 'settings' && (
            <div className="max-w-2xl mx-auto">
              <h2 className="text-xl font-bold mb-6 tracking-tight">Settings</h2>
              <div className={`rounded-2xl p-6 space-y-6 ${darkMode ? 'bg-slate-900 border border-slate-800' : 'bg-white shadow-sm border border-slate-100'}`}>
                
                <div className="flex items-center justify-between">
                  <div>
                    <p className="font-semibold text-sm">Dark Mode</p>
                    <p className={`text-xs ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>Cool down the interface for night viewing.</p>
                  </div>
                  <button 
                    onClick={() => setDarkMode(!darkMode)}
                    className={`w-12 h-6.5 rounded-full relative transition-colors ${darkMode ? 'bg-orange-500' : 'bg-slate-200'}`}
                  >
                    <div className={`absolute top-0.5 w-5.5 h-5.5 rounded-full bg-white shadow-sm transition-all ${darkMode ? 'left-6' : 'left-0.5'}`}></div>
                  </button>
                </div>

                <div className={`h-px ${darkMode ? 'bg-slate-800' : 'bg-slate-100'}`}></div>

                <div className="flex items-center justify-between">
                  <div>
                    <p className="font-semibold text-sm text-red-500">Wipe Log</p>
                    <p className={`text-xs ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>Permanently remove all logs from your burn history.</p>
                  </div>
                  <button 
                    onClick={clearAllHistory}
                    disabled={history.length === 0}
                    className={`px-3 py-1.5 rounded-lg font-bold text-[11px] uppercase tracking-wider transition-all ${history.length === 0 ? 'opacity-30 grayscale cursor-not-allowed border border-slate-200 text-slate-400' : 'bg-red-50 text-red-600 hover:bg-red-600 hover:text-white border border-red-100'}`}
                  >
                    Clear History
                  </button>
                </div>

                <div className={`h-px ${darkMode ? 'bg-slate-800' : 'bg-slate-100'}`}></div>

                <div>
                  <p className="font-semibold text-sm mb-1">About Fire AI</p>
                  <p className={`text-xs leading-relaxed ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>
                    Fire AI Music Engine v3.1.2<br/>
                    A high-performance melody discovery engine.
                  </p>
                </div>

              </div>
            </div>
          )}

          {activeTab === 'listen' && (
            <div className="h-full flex flex-col items-center justify-center max-w-lg mx-auto text-center">
              <div className="mb-12">
                <h2 className="text-3xl font-bold mb-2 tracking-tight">Melody Ignite</h2>
                <p className={`text-sm ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>Hum your tune to ignite the search engine.</p>
              </div>
              <div className="relative mb-14">
                {isListening && (
                  <div className="absolute -inset-10 flex items-center justify-center">
                    <div className="w-56 h-56 bg-orange-500/10 rounded-full animate-ping"></div>
                  </div>
                )}
                <button
                  onClick={isListening ? () => stopListening() : startListening}
                  className={`relative z-10 w-40 h-40 rounded-full flex flex-col items-center justify-center transition-all shadow-xl active:scale-95 ${isListening ? 'bg-orange-500 text-white' : `${darkMode ? 'bg-slate-800 text-orange-500 border-slate-700' : 'bg-white text-orange-500 border-2 border-orange-50'}`}`}
                >
                  {isListening ? <Activity size={48} className="animate-bounce" /> : <Flame size={48} />}
                  <span className="mt-4 text-[10px] font-bold uppercase tracking-widest">
                    {isListening ? 'Listening...' : 'Ignite Mic'}
                  </span>
                </button>
              </div>
              {isListening && (
                <div className="space-y-4 w-full">
                  <div className="flex flex-col items-center">
                    <span className="text-4xl font-bold text-orange-500 mb-1">{detectedNote || "--"}</span>
                    <span className={`text-[10px] font-mono uppercase tracking-widest ${darkMode ? 'text-slate-500' : 'text-slate-400'}`}>{detectedPitch || 0} Hz</span>
                  </div>
                </div>
              )}
            </div>
          )}
        </div>

        {/* Player Bar */}
        {currentTrack && (
          <div className={`fixed bottom-4 left-4 right-4 md:bottom-6 md:left-72 md:right-8 p-3 rounded-2xl flex items-center justify-between z-50 shadow-2xl animate-in slide-in-from-bottom-5 border ${darkMode ? 'bg-slate-900 border-slate-800' : 'bg-white border-slate-100 text-slate-900'}`}>
            <div className="absolute top-0 left-6 right-6 h-0.5 bg-slate-100/10 rounded-full overflow-hidden">
              <div className="h-full bg-orange-500 transition-all duration-300" style={{ width: `${progress}%` }}></div>
            </div>
            <div className="flex items-center w-1/3 min-w-0">
              <img src={currentTrack.artwork} className="w-9 h-9 rounded-lg mr-2 md:mr-3 object-cover shadow-sm" alt="" />
              <div className="min-w-0">
                <p className="text-[11px] md:text-xs font-semibold truncate">{currentTrack.title}</p>
                <p className="text-[9px] md:text-[10px] text-slate-400 truncate font-medium">{currentTrack.artist}</p>
              </div>
            </div>
            <div className="flex items-center">
              <button onClick={() => setIsPlaying(!isPlaying)} className={`p-2 md:p-2.5 rounded-full hover:scale-105 transition-transform active:scale-95 shadow-sm ${darkMode ? 'bg-white text-slate-900' : 'bg-orange-500 text-white'}`}>
                {isPlaying ? <Pause size={18} fill="currentColor" /> : <Play size={18} fill="currentColor" className="ml-0.5" />}
              </button>
            </div>
            <div className="w-1/3 flex justify-end items-center space-x-1 md:space-x-3">
              <a href={getYouTubeLink(currentTrack)} target="_blank" rel="noopener noreferrer" className="p-2 hover:bg-slate-100 rounded-lg text-slate-400 hover:text-red-500 transition-all hidden sm:block">
                <Youtube size={18} />
              </a>
              <button onClick={() => {setCurrentTrack(null); setIsPlaying(false);}} className="p-2 hover:bg-slate-100 rounded-lg transition-colors">
                <X size={18} className="text-slate-400" />
              </button>
            </div>
          </div>
        )}
      </main>
    </div>
  );
}
